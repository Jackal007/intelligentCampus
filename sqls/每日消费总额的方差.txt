SELECT
	sum(item) / days
FROM
	(
		SELECT
			pow((sum(deal_cost) - (total_cost/days)),2) AS item,
			days as days
		FROM
			card,
			(
				SELECT
					count(date(deal_date)) AS days,
					sum(deal_cost)  as total_cost
				FROM
					card
				WHERE
					student_id = 3346
			) AS a
		WHERE
			student_id = 3346
		GROUP BY
			date(deal_date)
	) AS b
